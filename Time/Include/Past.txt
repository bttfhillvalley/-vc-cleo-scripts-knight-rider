:CheckPastTravel
// Check to see if another time machine has travelled and remove if we go to the past
// Currently no good way of telling when to get the car back due to lack of variables
// needed to tell when the time machine reentered.  Might have to do a cleo_call and rewrite more.
0AB4: var1 = var TRAVEL
if
    var1 == 1
then
    0AB4: var1 = var CDATE
    0AB4: var2 = var CTIME
    
    0AB4: var3 = var SET_TIME
    
    while var3 == 0
        wait 0
        0AB4: var3 = var SET_TIME
    end
    while var3 == 1
        wait 0
        0AB4: var3 = var SET_TIME
    end
    0AB4: var3 = var CDATE
    0AB4: var3 = var CTIME
    
    if
        var3 < var1
    then  
        gosub @StoreVehicleInfo 
        Car.Destroy(vehicle)        
        terminate_this_custom_script
        cleo_return 1 true
    end
    
    if and
        var3 == var1
        var4 < var2
    then
        gosub @StoreVehicleInfo   
        Car.Destroy(vehicle)
        terminate_this_custom_script
        cleo_return 1 true
    end
    
end
cleo_return 1 false

:TimePast
while true
    // Wait until we hit the target time
    wait 0
    0AB4: 2@ = var CDATE
    0AB4: 3@ = var CTIME
    0AB4: 4@ = var SET_TIME

    if and
        003C:   2@ == 0@ // Same Date
        003C:   3@ == 1@ // Same Time
    then
        repeat
            wait 0
            0AB4: 5@ = var CFADE
        until 5@ == 0
        0AB4: 7@ = var MODE

        if or
            6@ == 0 // Not travelling?
            7@ == 0 // Cutscene mode
        then
            // Play Doc's beeper
            3F84: play_sound "doc_watch_beep.wav" loop 0
            wait 750
            cleo_return 1 NORMAL_REENTRY  // Do normal reentry
        else
            cleo_return 1 INST_REENTRY // Do Instant reentry
        end
    else
        // Only preload the time machine in if we are changing time periods.
        if
            4@ == 1 // Setting time right now
        then
            repeat
                wait 0
                0AB4: 4@ = var SET_TIME
            until 4@ == 0
            0AB4: 2@ = var CDATE
            0AB4: 3@ = var CTIME
            // Check same date, but time is in the future
            if and
                003C:   2@ == 0@ // Same Date
                001E:   3@ > 1@  // Greater Time
            then
                cleo_return 1 PRELOAD_REENTRY // flag to spawn time machine immediately
            else if
                001E:   2@ > 0@ // Greater Date
            then
                cleo_return 1 PRELOAD_REENTRY // flag to spawn time machine immediately
            end
            end
        end
    end
end
