:TimeCoilDelorean
if or
    var1 == 0 // Non-hoodbox version
    vehicle_speed > 47.0 // floating-point values
then
    gosub @CalculateSpeedOffset
    gosub @ShowCoils1
    gosub @ShowCoils2
else if
    var4 == 0
then
    gosub @CalculateSpeedOffset
    gosub @ShowCoils1
    gosub @ShowCoils2
else if
    var4 == 1
then
    gosub @CalculateSpeedOffset
    gosub @ShowCoils1
    gosub @HideCoils2
else
    gosub @CalculateSpeedOffset
    gosub @HideCoils1
    gosub @HideCoils2
end
end
end
return
