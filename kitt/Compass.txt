{$INCLUDE variables.txt}
0000:
int vehicle
int Turning_Radius
int step10


{ how to write a message on kitts message board 
10 blocks Left to right 
"Message" is the name of the frame then character, lastly block
36 total characters, spaces are named blank, (five)5 = letter s , (Zero)0 = letter o
example "Message05" would place a 0 in the 5th block from the left 
be sure to hide the blanks before putting a message in!
}


while true
    wait 10
    if
        Car.Wrecked(vehicle)
    then
        terminate_this_custom_script
    end
    0174: Turning_Radius = car vehicle z_angle
    008C: Turning_Radius = float_to_integer Turning_Radius
    3F02: Engine_IO = car vehicle engine_status
    if
        Turning_Radius > 360
    then 
        Turning_Radius = 360 
    end    

    if
    Engine_IO == 0
    then
    gosub @ClearMessage
    else
  
    if or
        Turning_Radius == 360
         Turning_Radius == 0
    then
       gosub @ClearMessage
       wait 0
       gosub @NorthMessage
    else 
           if
        Turning_Radius == 180
    then
       gosub @ClearMessage
       wait 0
       gosub @SouthMessage
       else

         
            if
        Turning_Radius == 270
    then
       gosub @ClearMessage
       wait 0
       gosub @WestMessage
       else
    
        if
        Turning_Radius == 90
    then
       gosub @ClearMessage
       wait 0
       gosub @EastMessage    
      else 
           if
        Turning_Radius == 315
    then
       gosub @ClearMessage
       wait 0
       gosub @NorthWestMessage   
       else 
                  if
        Turning_Radius == 45
    then
       gosub @ClearMessage
       wait 0
       gosub @NorthEastMessage    
       else 
                  if
        Turning_Radius == 225
    then
       gosub @ClearMessage
       wait 0
       gosub @SouthWestMessage   
       else 
                  if
        Turning_Radius == 135
    then
       gosub @ClearMessage
       wait 0
       gosub @SouthEastMessage   
   
    end
    end 
    end   
    end    
    end 
    end
    end   
    end 
    end
    end
return
